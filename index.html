OpenAI API调用示例（Node.js）


// OpenAI API通过4Z API中转调用示例
const axios = require('axios');

const API_KEY = 'your_4Z_api_key'; // 替换为您的4Z API密钥
const API_URL = 'https://api.zzzzapi.com/v1/chat/completions';

async function callOpenAI() {
  try {
    const response = await axios.post(
      API_URL,
      {
        model: 'gpt-4o', // 支持所有OpenAI模型
        messages: [
          { role: 'system', content: '你是由OpenAI开发的AI助手' },
          { role: 'user', content: '请详细介绍OpenAI API的主要功能和应用场景' }
        ],
        temperature: 0.7,
        max_tokens: 2000
      },
      {
        headers: {
          'Authorization': `Bearer ${API_KEY}`,
          'Content-Type': 'application/json'
        }
      }
    );

    console.log(JSON.stringify(response.data, null, 2));
  } catch (error) {
    console.error('调用OpenAI API时出错:', error.response?.data || error.message);
  }
}

callOpenAI();
